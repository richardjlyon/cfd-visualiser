       ```python
       """Site-wide Open Graph social-card PNG (EDIT-02 + UI-SPEC OG:image Spec)."""
       from __future__ import annotations
       import json
       from pathlib import Path
       import matplotlib
       matplotlib.use("Agg")
       import matplotlib.pyplot as plt

       OKABE_BLUE = "#0072B2"
       OKABE_ORANGE = "#E69F00"

       def build(chart_json_path: Path | str, out_path: Path | str) -> Path:
           rows = json.loads(Path(chart_json_path).read_text())
           # Aggregate to 'All' allocation rounds for the social card
           by_month = {}
           for r in rows:
               m = r["month"]
               agg = by_month.setdefault(m, {"sw": 0, "mw": 0, "g": 0})
               agg["sw"] += r["strike"] * r["generation_mwh"]
               agg["mw"] += r["market"] * r["generation_mwh"]
               agg["g"] += r["generation_mwh"]
           months = sorted(by_month)
           strike = [by_month[m]["sw"] / by_month[m]["g"] for m in months]
           market = [by_month[m]["mw"] / by_month[m]["g"] for m in months]

           fig, ax = plt.subplots(figsize=(12, 6.3), dpi=100)
           ax.plot(months, strike, color=OKABE_BLUE, linewidth=2.5,
                   label="Strike price")
           ax.plot(months, market, color=OKABE_ORANGE, linewidth=2.5,
                   label="Market reference price")
           ax.set_ylabel("£ / MWh", fontsize=14)
           ax.set_title("CfD Visualiser", loc="left", fontsize=22,
                        fontweight="semibold")
           ax.set_title("Consumers paid the gap in every year except 2022.",
                        loc="right", fontsize=12)
           ax.grid(alpha=0.3)
           ax.legend(loc="upper left", fontsize=12)
           # Show every 12th month tick for readability
           tick_idx = list(range(0, len(months), 12))
           ax.set_xticks(tick_idx)
           ax.set_xticklabels([months[i] for i in tick_idx], rotation=0)
           fig.tight_layout()
           Path(out_path).parent.mkdir(parents=True, exist_ok=True)
           fig.savefig(out_path, dpi=100, format="png")
           plt.close(fig)
           return Path(out_path)
       ```

       ## Failure Playbook

       | Symptom | Cause | Action |
       |---------|-------|--------|
       | Healthchecks alert email | Daily run missed or failed | Check GitHub Actions daily-rebuild log; look at last exit code |
       | Exit 1 in pipeline | Fetch failed (LCCC URL 5xx/404/HTML) | Check LCCC portal status; if URL changed, update `pipeline/fetch.py::LCCC_URL` |

       ## Keepalive — GitHub 60-day Inactivity Rule

       successful daily run commits `data/raw/YYYY-MM-DD.csv.gz` (plus

       resets the 60-day clock. No separate `last_build.txt` touch file is
       needed, and none is created — the data archive itself *is* the
       activity signal.

       but possible):** Healthchecks.io will alert long before the 60-day
       mark. If the workflow is ever auto-disabled, run
       `gh workflow enable daily.yml` to re-enable it, then
       `gh workflow run daily.yml` to trigger a manual run. Document any
       such incident in an OPS incident log.
